102/*
103** Any value being manipulated by the program either is non
104** collectable, or the collectable object has the right tag
105** and it is not dead. The option 'L == NULL' allows other
106** macros using this one to be used where L is not available.
107*/
108#define checkliveness(L,obj) \
109 ((void)L, lua_longassert(!iscollectable(obj) || \
110 (righttt(obj) && (L == NULL || !isdead(G(L),gcvalue(obj))))))

728/*
729** About 'alimit': if 'isrealasize(t)' is true, then 'alimit' is the
730** real size of 'array'. Otherwise, the real size of 'array' is the
731** smallest power of two not smaller than 'alimit' (or zero iff 'alimit'
732** is zero); 'alimit' is then used as a hint for #t.
733*/
734
735#define BITRAS (1 << 7)
736#define isrealasize(t) (!((t)->flags & BITRAS))
737#define setrealasize(t) ((t)->flags &= cast_byte(~BITRAS))
