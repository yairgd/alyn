397@@ over a floating number.
398@@ l_floatatt(x) corrects float attribute 'x' to the proper float type
399** by prefixing it with one of FLT/DBL/LDBL.
400@@ LUA_NUMBER_FRMLEN is the length modifier for writing floats.
401@@ LUA_NUMBER_FMT is the format for writing floats.
402@@ lua_number2str converts a float to a string.
403@@ l_mathop allows the addition of an 'l' or 'f' to all math operations.
404@@ l_floor takes the floor of a float.

416/*
417@@ lua_numbertointeger converts a float number with an integral value
418** to an integer, or returns 0 if float is not within the range of
419** a lua_Integer. (The range comparisons are tricky because of
420** rounding. The tests here assume a two-complement representation,
421** where MININTEGER always has an exact representation as a float;
422** MAXINTEGER may not have one, and therefore its conversion to float
423** may have an ill-defined value.)
424*/
425#define lua_numbertointeger(n,p) \
426 ((n) >= (LUA_NUMBER)(LUA_MININTEGER) && \
427 (n) < -(LUA_NUMBER)(LUA_MININTEGER) && \
428 (*(p) = (LUA_INTEGER)(n), 1))
