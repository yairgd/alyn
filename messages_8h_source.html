67 void Serialize() {
68 int n = 0;
69 crc = 0;
70 buffer [n++] = preamble;
71 buffer [n++] = 123;//type;
72 buffer [n++] = static_cast<char>(cmd);
73 buffer [n++] = sizeof (MsgData);
74 if (sizeof (MsgData)) {
75 std::memcpy (&buffer[n], &data, sizeof (MsgData));
76 n+=sizeof (MsgData);
77 }
78 buffer [n++] = 0; // empty crc to calculate later
79 buffer [n++] = postamble;
80
81 // find crc
82 for (int i = 0;i < (int)size();i++) {
83 auto c = static_cast<unsigned char>(buffer[i]);
84 crc += c;
85 }
86 buffer[n-2] = crc;
87 }
